32 // The expected background level for the envelope service
33 // This parameter is not expected to change as long as noise level normalization is active
34 #define ENVELOPE_BACKGROUND_LEVEL 100
35 
36 // Parameters for direct leakage subtraction
37 // To be set from inspection of the worst case direct leakage
38 // LEAKAGE_SAMPLE_POSITION_M and LEAKAGE_END_POSITION_M must be within
39 // the measurement range [RANGE_START_M, RANGE_START_M + RANGE_LENGTH_M]
40 // and LEAKAGE_END_POSITION_M must be beyond LEAKAGE_SAMPLE_POSITION_M
41 #define LEAKAGE_SAMPLE_POSITION_M 0.15f
42 #define LEAKAGE_END_POSITION_M 0.30f
43 #define MAX_LEAK_AMPLITUDE 2000
44 
45 // The number of observations in the parking detection queue
46 #define DETECTION_OBSERVATION_COUNT 3
47 
48 // The minimal weight for each observation in the parking detection queue for
49 // detection of a parked car
50 #define DETECTION_WEIGHT_THRESHOLD 5.0f
51 
52 // Parameters for exclusion of transient reflections from people and items that are near the
53 // sensor for short durations
54 #define DETECTION_WEIGHT_RATIO_LIMIT 3.0f
55 #define DETECTION_DISPLACEMENT_LIMIT 0.1f
56 
57 // The time duration between two consecutive sweeps
58 #define DETECTOR_SWEEP_PERIOD_S 10.0f
59 
60 // Minimal envelope service runtime before sensor recalibration due to a data quality warning.
61 // A sensor calibration is costly in terms of power consumption relative to the sweeps
62 #define SERVICE_RUNTIME_MIN_S 900.0f
63 
64 // Maximal envelope service active time before the service is destroyed and recreated
65 // to renew the noise level normalization. The sensor output amplitude may change notably due
66 // to temperature changes before a data quality warning is triggered. This type of refresh
67 // is inexpensive with respect to power consumption.
68 #define SERVICE_UPTIME_MAX_S 900.0f

123 int acc_ref_app_parking(int argc, char *argv[])
124 {
125  (void)argc;
126  (void)argv;
127  printf("Acconeer software version %s\n", acc_version_get());
128 
129  const acc_hal_t *hal = acc_hal_integration_get_implementation();
130 
131  if (!acc_rss_activate(hal))
132  {
133  printf("Failed to activate RSS\n");
134  return EXIT_FAILURE;
135  }
136 
137  acc_service_configuration_t configuration = acc_service_envelope_configuration_create();
138 
139  if (configuration == NULL)
140  {
141  printf("Failed to create service configuration\n");
142  acc_rss_deactivate();
143  return EXIT_FAILURE;
144  }
145 
146  if (!sensor_calibration())
147  {
148  printf("Failed to calibrate sensor\n");
149  acc_service_envelope_configuration_destroy(&configuration);
150  acc_rss_deactivate();
151  return EXIT_FAILURE;
152  }
153 
154  configure_service(configuration);
155 
156  acc_service_handle_t handle = acc_service_create(configuration);
157 
158  if (handle == NULL)
159  {
160  printf("Failed to create service handle\n");
161  acc_service_envelope_configuration_destroy(&configuration);
162  acc_rss_deactivate();
163  return EXIT_FAILURE;
164  }
165 
166  acc_service_envelope_metadata_t metadata;
167  acc_service_envelope_get_metadata(handle, &metadata);
168 
169  uint16_t leak_sample_index = (uint16_t)(((LEAKAGE_SAMPLE_POSITION_M - metadata.start_m) / metadata.step_length_m) + 0.5f);
170  uint16_t leak_end_index = (uint16_t)(((LEAKAGE_END_POSITION_M - metadata.start_m) / metadata.step_length_m) + 0.5f);
171  bool valid_leak_setup = (leak_sample_index < leak_end_index && leak_sample_index < metadata.data_length);
172 
173  uint16_t *data = NULL;
174  acc_service_envelope_result_info_t result_info;
175  sweep_observable_t observations[DETECTION_OBSERVATION_COUNT];
176  uint16_t observation_count = 0;
177  uint32_t last_update_ms = 0;
178  uint32_t last_activate_ms = hal->os.gettime();
179  uint32_t last_calibration_ms = hal->os.gettime();
180  uint16_t sweep_index = 0;
181 
182  bool status = true;
183 
184  if (!valid_leak_setup)
185  {
186  printf("Parameters are not valid\n");
187  status = false;
188  }
189  else
190  {
191  status = acc_service_activate(handle);
192  }
193 
194  while (status)
195  {
196  if (hal->os.gettime() - last_activate_ms > SERVICE_UPTIME_MAX_S * 1000)
197  {
198  status = service_recreate(configuration, &handle);
199  last_activate_ms = hal->os.gettime();
200  }
201 
202  if (status)
203  {
204  while (last_update_ms != 0 && hal->os.gettime() - last_update_ms < DETECTOR_SWEEP_PERIOD_S * 1000)
205  {
206  acc_integration_sleep_ms((DETECTOR_SWEEP_PERIOD_S * 1000) - (hal->os.gettime() - last_update_ms));
207  }
208 
209  status = acc_service_envelope_get_next_by_reference(handle, &data, &result_info);
210  last_update_ms = hal->os.gettime();
211  }
212 
213  if (status && result_info.data_quality_warning &&
214  hal->os.gettime() - last_calibration_ms > SERVICE_RUNTIME_MIN_S * 1000)
215  {
216  status = acc_service_deactivate(handle);
217 
218  if (status)
219  {
220  acc_service_destroy(&handle);
221 
222  status = sensor_calibration();
223  }
224 
225  if (status)
226  {
227  handle = acc_service_create(configuration);
228 
229  status = handle != NULL;
230  }
231 
232  if (status)
233  {
234  status = acc_service_activate(handle);
235  }
236 
237  last_calibration_ms = hal->os.gettime();
238 
239  if (status)
240  {
241  status = acc_service_envelope_get_next_by_reference(handle, &data, &result_info);
242  last_update_ms = hal->os.gettime();
243  }
244  }
245 
246  if (status)
247  {
248  bool detection = parking_detection(&metadata, leak_sample_index, leak_end_index, observations, &observation_count, data);
249 
250  if (sweep_index < DETECTION_OBSERVATION_COUNT - 1)
251  {
252  printf("%" PRIu16 ": No result\n", sweep_index);
253  }
254  else
255  {
256  printf("%" PRIu16 ": %s\n", sweep_index, detection ? "Car" : "No car");
257  }
258 
259  sweep_index++;
260  }
261  }
262 
263  acc_service_envelope_configuration_destroy(&configuration);
264  acc_service_deactivate(handle);
265  acc_service_destroy(&handle);
266  acc_rss_deactivate();
267 
268  return status ? EXIT_SUCCESS : EXIT_FAILURE;
269 }

273 {
274  if (!acc_service_deactivate(*handle))
275  {
276  return false;
277  }
278 
279  acc_service_destroy(handle);
280 
281  *handle = acc_service_create(configuration);
282 
283  if (*handle == NULL)
284  {
285  return false;
286  }
287 
288  if (!acc_service_activate(*handle))
289  {
290  return false;
291  }
292 
293  return true;
294 }

327 bool parking_detection(const acc_service_envelope_metadata_t *metadata, uint16_t leak_sample_index,
328  uint16_t leak_end_index, sweep_observable_t *observations, uint16_t *observation_count,
329  const uint16_t *data)
330 {
331  float weight_sum = 0.0f;
332  float weight_sum_r = 0.0f;
333  float leak_start = 0.0f;
334  uint16_t leak_amplitude = MAX_LEAK_AMPLITUDE < data[leak_sample_index] ? MAX_LEAK_AMPLITUDE : data[leak_sample_index];
335  uint16_t a_leak = leak_amplitude < ENVELOPE_BACKGROUND_LEVEL ? 0 : leak_amplitude - ENVELOPE_BACKGROUND_LEVEL;
336  float leak_step = ((float)(a_leak) / (leak_end_index - leak_sample_index));
337 
338  leak_start = leak_end_index * leak_step + ENVELOPE_BACKGROUND_LEVEL;
339 
340  for (uint16_t i = 0; i < metadata->data_length; i++)
341  {
342  float r = metadata->start_m + i * metadata->step_length_m;
343  float bg = 0.0f;
344  if (i <= leak_end_index)
345  {
346  bg = leak_start - i * leak_step;
347  }
348  else
349  {
350  bg = ENVELOPE_BACKGROUND_LEVEL;
351  }
352 
353  float sweep_above_bg = data[i] - bg;
354  sweep_above_bg = sweep_above_bg > 0.0f ? sweep_above_bg : 0.0f;
355 
356  float weight = sweep_above_bg / ENVELOPE_BACKGROUND_LEVEL;
357  weight = weight < 1.0f ? weight : 1.0f;
358  weight = weight * sweep_above_bg * r;
359 
360  weight_sum += weight;
361  weight_sum_r += weight * r;
362  }
363 
364  if (*observation_count == DETECTION_OBSERVATION_COUNT)
365  {
366  for (uint16_t i = 1; i < DETECTION_OBSERVATION_COUNT; i++)
367  {
368  observations[i - 1].weight = observations[i].weight;
369  observations[i - 1].distance = observations[i].distance;
370  }
371  }
372  else
373  {
374  (*observation_count)++;
375  }
376 
377  observations[*observation_count - 1].weight = weight_sum / metadata->data_length;
378  observations[*observation_count - 1].distance = weight_sum_r / weight_sum;
379 
380  float weight_min = __FLT_MAX__;
381  float weight_max = 0.0f;
382  float distance_min = __FLT_MAX__;
383  float distance_max = 0.0f;
384 
385  for (uint16_t i = 0; i < *observation_count; i++)
386  {
387  if (weight_min > observations[i].weight)
388  {
389  weight_min = observations[i].weight;
390  }
391 
392  if (weight_max < observations[i].weight)
393  {
394  weight_max = observations[i].weight;
395  }
396 
397  if (distance_min > observations[i].distance)
398  {
399  distance_min = observations[i].distance;
400  }
401 
402  if (distance_max < observations[i].distance)
403  {
404  distance_max = observations[i].distance;
405  }
406  }
407 
408  bool detection = *observation_count == DETECTION_OBSERVATION_COUNT &&
409  weight_min >= DETECTION_WEIGHT_THRESHOLD &&
410  weight_max / weight_min <= DETECTION_WEIGHT_RATIO_LIMIT &&
411  distance_max - distance_min <= DETECTION_DISPLACEMENT_LIMIT;
412 
413  return detection;
414 }

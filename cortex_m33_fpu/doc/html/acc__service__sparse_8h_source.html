22 /**
23  * @brief Sampling mode
24  *
25  * The sampling mode changes how the hardware accelerated averaging is done. Mode A is optimized
26  * for maximal independence of the depth points, giving a higher depth resolution than mode B.
27  * Mode B is instead optimized for maximal radar loop gain per unit time spent on measuring. This
28  * makes it more energy efficient and suitable for cases where small movements are to be detected
29  * over long ranges. Mode A is more suitable for applications like gesture recognition, measuring
30  * the distance to a movement, and speed measurements.
31  *
32  * Mode B typically gives roughly 3 dB better SNR per unit time than mode A. However, please note
33  * that very short ranges of only one or a few points are suboptimal with mode B. In those cases,
34  * always use mode A.
35  */
36 typedef enum
37 {
38  ACC_SERVICE_SPARSE_SAMPLING_MODE_A,
39  ACC_SERVICE_SPARSE_SAMPLING_MODE_B
40 } acc_service_sparse_sampling_mode_enum_t;
41 typedef uint32_t acc_service_sparse_sampling_mode_t;

163 /**
164  * @brief Get the sensor downsampling factor
165  *
166  * Gets the downsampling factor - the number of steps taken between each data point. A downsampling factor of 1 samples
167  * every possible point in the range. A downsampling factor of 2 samples every other point, and so on.
168  *
169  * In the sparse service, the base step length is ~6cm. Thus, for example setting downsampling factor to 4 makes
170  * the distance between two points in the measured range ~24cm.
171  *
172  * @param[in] service_configuration The configuration to get downsampling factor from
173  * @return The downsampling factor
174  */
175 uint16_t acc_service_sparse_downsampling_factor_get(acc_service_configuration_t service_configuration);

141 static bool run_test(acc_rss_assembly_test_configuration_t configuration)
142 {
143  acc_rss_assembly_test_result_t test_results[ACC_RSS_ASSEMBLY_TEST_MAX_NUMBER_OF_TESTS];
144  uint16_t nr_of_test_results = ACC_RSS_ASSEMBLY_TEST_MAX_NUMBER_OF_TESTS;
145 
146  bool success = acc_rss_assembly_test(configuration, test_results, &nr_of_test_results);
147 
148  if (success)
149  {
150  for (uint16_t i = 0; i < nr_of_test_results; i++)
151  {
152  const bool passed = test_results[i].test_passed;
153  printf("Name: %s, result: %s\n", test_results[i].test_name, passed ? "Pass" : "Fail");
154  }
155  }
156  else
157  {
158  printf("Bring up test: Failed to complete\n");
159  return success;
160  }
161 
162  return success;
163 }

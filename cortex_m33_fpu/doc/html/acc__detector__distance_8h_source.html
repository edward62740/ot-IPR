89 /**
90  * @brief Enum for peak sorting algorithms
91  */
92 typedef enum
93 {
94  /* Return peaks with the closest detection first. */
95  ACC_DETECTOR_DISTANCE_PEAK_SORTING_CLOSEST_FIRST,
96  /* Return peaks with the peak with the highest amplitude first. */
97  ACC_DETECTOR_DISTANCE_PEAK_SORTING_STRONGEST_FIRST,
98  /* Return peaks with the peak from the strongest reflector first.
99  The decrease in amplitude over distance is accounted for.
100  Cannot be combined with negative start range. */
101  ACC_DETECTOR_DISTANCE_PEAK_SORTING_STRONGEST_REFLECTOR_FIRST,
102  /* Return peaks with the peak from the strongest flat reflector first.
103  The decrease in amplitude over distance is accounted for.
104  Cannot be combined with negative start range. */
105  ACC_DETECTOR_DISTANCE_PEAK_SORTING_STRONGEST_FLAT_REFLECTOR_FIRST
106 } acc_detector_distance_peak_sorting_t;

22  * @n

24  * - Activate Radar System Software (RSS)
25  * - Create a Power Bins service configuration
26  * - Create a Power Bins service using the previously created configuration
27  * - Destroy the Power Bins service configuration
28  * - Activate the Power Bins service

30  * - Deactivate and destroy the Power Bins service
31  * - Deactivate Radar System Software (RSS)
32  */

44 int acc_example_service_power_bins(int argc, char *argv[])
45 {
46  (void)argc;
47  (void)argv;
48  printf("Acconeer software version %s\n", acc_version_get());
49 
50  const acc_hal_t *hal = acc_hal_integration_get_implementation();
51 
52  if (!acc_rss_activate(hal))
53  {
54  printf("acc_rss_activate() failed\n");
55  return EXIT_FAILURE;
56  }
57 
58  acc_service_configuration_t power_bins_configuration = acc_service_power_bins_configuration_create();
59 
60  if (power_bins_configuration == NULL)
61  {
62  printf("acc_service_power_bins_configuration_create() failed\n");
63  acc_rss_deactivate();
64  return EXIT_FAILURE;
65  }
66 
67  update_configuration(power_bins_configuration);
68 
69  acc_service_handle_t handle = acc_service_create(power_bins_configuration);
70 
71  if (handle == NULL)
72  {
73  printf("acc_service_create() failed\n");
74  acc_service_power_bins_configuration_destroy(&power_bins_configuration);
75  acc_rss_deactivate();
76  return EXIT_FAILURE;
77  }
78 
79  acc_service_power_bins_configuration_destroy(&power_bins_configuration);
80 
81  acc_service_power_bins_metadata_t power_bins_metadata = { 0 };
82  acc_service_power_bins_get_metadata(handle, &power_bins_metadata);
83 
84  printf("Start: %d mm\n", (int)(power_bins_metadata.start_m * 1000.0f));
85  printf("Length: %u mm\n", (unsigned int)(power_bins_metadata.length_m * 1000.0f));
86  printf("Bin count: %u\n", (unsigned int)(power_bins_metadata.bin_count));
87 
88  if (!acc_service_activate(handle))
89  {
90  printf("acc_service_activate() failed\n");
91  acc_service_destroy(&handle);
92  acc_rss_deactivate();
93  return EXIT_FAILURE;
94  }
95 
96  bool success = true;
97  const int iterations = 5;
98  uint16_t data[power_bins_metadata.bin_count];
99  acc_service_power_bins_result_info_t result_info;
100 
101  for (int i = 0; i < iterations; i++)
102  {
103  success = acc_service_power_bins_get_next(handle, data, power_bins_metadata.bin_count, &result_info);
104 
105  if (!success)
106  {
107  printf("acc_service_power_bins_get_next() failed\n");
108  break;
109  }
110 
111  print_data(data, power_bins_metadata.bin_count);
112  }
113 
114  bool deactivated = acc_service_deactivate(handle);
115 
116  acc_service_destroy(&handle);
117 
118  acc_rss_deactivate();
119 
120  if (deactivated && success)
121  {
122  printf("Application finished OK\n");
123  return EXIT_SUCCESS;
124  }
125 
126  return EXIT_FAILURE;
127 }

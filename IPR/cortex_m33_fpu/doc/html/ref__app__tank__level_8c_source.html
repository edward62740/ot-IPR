208 int acc_ref_app_tank_level(int argc, char *argv[])
209 {
210  (void)argc;
211  (void)argv;
212  printf("Acconeer software version %s\n", acc_version_get());
213 
214  const acc_hal_t *hal = acc_hal_integration_get_implementation();
215 
216  if (!acc_rss_activate(hal))
217  {
218  printf("Failed to activate RSS\n");
219  return EXIT_FAILURE;
220  }
221 
222  acc_detector_distance_configuration_t distance_configuration = acc_detector_distance_configuration_create();
223 
224  if (distance_configuration == NULL)
225  {
226  printf("Failed to create detector configuration\n");
227  acc_rss_deactivate();
228  return EXIT_FAILURE;
229  }
230 
231  acc_detector_distance_handle_t distance_handle = acc_detector_distance_create(distance_configuration);
232 
233  if (distance_handle == NULL)
234  {
235  printf("Failed to create detector\n");
236  acc_detector_distance_configuration_destroy(&distance_configuration);
237  acc_rss_deactivate();
238  return EXIT_FAILURE;
239  }
240 
241  if (!record_backgrounds(&distance_handle, distance_configuration))
242  {
243  printf("Failed to calibrate detector\n");
244  acc_detector_distance_configuration_destroy(&distance_configuration);
245  acc_detector_distance_destroy(&distance_handle);
246  acc_rss_deactivate();
247  return EXIT_FAILURE;
248  }
249 
250  // If there is a long time between measurements it is good to calibrate the sensor
251  // before each measurement
252  acc_detector_distance_destroy(&distance_handle);
253 
254  if (!sensor_calibration())
255  {
256  printf("Failed to calibrate sensor\n");
257  return EXIT_FAILURE;
258  }
259 
260  distance_handle = acc_detector_distance_create(distance_configuration);
261 
262  bool status = true;
263 
264  while (status)
265  {
266  float distance = 0.0f;
267  bool distance_detected = false;
268 
269  printf("Measure close range\n");
270  status = measure_close_range(&distance_handle, distance_configuration, &distance_detected, &distance);
271  if (status && !distance_detected)
272  {
273  printf("Measure mid range\n");
274  status = measure_mid_range(&distance_handle, distance_configuration, &distance_detected, &distance);
275  }
276 
277  if (status && !distance_detected)
278  {
279  printf("Measure far range\n");
280  status = measure_far_range(&distance_handle, distance_configuration, &distance_detected, &distance);
281  }
282 
283  if (!status)
284  {
285  printf("Measurement failed\n");
286  break;
287  }
288 
289  if (distance_detected)
290  {
291  printf("Peak at %u mm\n", (unsigned int)(distance * 1000));
292  }
293  else
294  {
295  printf("No peak found\n");
296  }
297 
298  //Add a call to a sleep function here to limit measurement update rate
299  }
300 
301  acc_detector_distance_configuration_destroy(&distance_configuration);
302  acc_detector_distance_destroy(&distance_handle);
303  acc_rss_deactivate();
304 
305  return status ? EXIT_SUCCESS : EXIT_FAILURE;
306 }

408 bool record_background(acc_detector_distance_handle_t *distance_handle,
409  acc_detector_distance_configuration_t distance_configuration,
410  float *range_gain, uint16_t *background, uint16_t background_length)
411 {
412  acc_detector_distance_recorded_background_info_t recorded_background_info;
413  bool status = true;
414  bool previous_gain_saturated = true;
415 
416  float gain = acc_detector_distance_configuration_receiver_gain_get(distance_configuration);
417 
418  recorded_background_info.data_saturated = true;
419 
420  while (status && gain > 0)
421  {
422  acc_detector_distance_configuration_receiver_gain_set(distance_configuration, gain);
423 
424  status = acc_detector_distance_reconfigure(distance_handle, distance_configuration);
425 
426  if (status)
427  {
428  status = acc_detector_distance_record_background(*distance_handle, background, background_length,
429  &recorded_background_info);
430 
431  if (!status || (!previous_gain_saturated && !recorded_background_info.data_saturated))
432  {
433  break;
434  }
435 
436  previous_gain_saturated = recorded_background_info.data_saturated;
437 
438  gain = gain - GAIN_STEP;
439  }
440  }
441 
442  if (status && recorded_background_info.data_saturated)
443  {
444  printf("Unable to record background without data saturation\n");
445  status = false;
446  }
447 
448  *range_gain = gain;
449 
450  return status;
451 }

502  acc_detector_distance_configuration_t distance_configuration,
503  acc_detector_distance_result_t *result,
504  acc_detector_distance_result_info_t *result_info)
505 {
506  float gain = acc_detector_distance_configuration_receiver_gain_get(distance_configuration);
507 
508  result_info->data_saturated = true;
509 
510  while (gain > 0)
511  {
512  if (!acc_detector_distance_activate(*distance_handle))
513  {
514  printf("Failed to activate detector\n");

517

524  if (!acc_detector_distance_deactivate(*distance_handle))
525  {
526  printf("Failed to deactivate detector\n");
527  return false;
528  }
529 
530  if (!result_info->data_saturated || gain < GAIN_STEP)
531  {
532  break;
533  }
534 
535  gain = gain - GAIN_STEP;
536 
537  acc_detector_distance_configuration_receiver_gain_set(distance_configuration, gain);
538 
539  if (!acc_detector_distance_reconfigure(distance_handle, distance_configuration))
540  {
541  return false;
542  }
543  }
544 
545  if (result_info->data_saturated)
546  {
547  printf("Unable to measure without data saturation\n");
548  return false;
549  }
550 
551  return true;
552 }

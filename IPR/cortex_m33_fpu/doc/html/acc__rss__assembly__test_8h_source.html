182 /**
183  * @brief Run assembly test
184  *
185  * This function executes a suite of tests suitable for PCB assembly testing.
186  *
187  * @details The caller need to allocate an array of assembly test results of at least the size
188  * of ACC_RSS_ASSEMBLY_TEST_MAX_NUMBER_OF_TESTS. The array size is to be provided in the
189  * nr_of_test_results parameter. If the size is not sufficiently large the test will
190  * fail.
191  *
192  * If a test fails to execute the return value will be set to false.
193  * If a test cannot pass its test limits the test_passed member of test_results will be
194  * set to false but the return value of this function will still be true.
195  *
196  * @param[in] test_configuration The test configuration
197  * @param[out] test_results Reference to struct where the test results will be stored
198  * @param[in,out] nr_of_test_results Input is the maximum number of items in the results array.
199  * Output is the actual number of test results available after
200  * the execution of the tests.
201  * @return True if successfully run, false otherwise
202  */
203 bool acc_rss_assembly_test(acc_rss_assembly_test_configuration_t test_configuration, acc_rss_assembly_test_result_t *test_results,
204  uint16_t *nr_of_test_results);

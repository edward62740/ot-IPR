57 /**
58  * @brief Use the presence detector to detect movement with low power
59  *
60  * @param[in] handle The presence detector handle
61  */
62 static bool execute_wakeup(acc_detector_presence_handle_t handle)
63 {
64  acc_detector_presence_result_t result;
65 
66  if (!acc_detector_presence_activate(handle))
67  {
68  printf("Failed to activate detector\n");
69  return false;
70  }
71 
72  do
73  {
74  if (!acc_detector_presence_get_next(handle, &result))
75  {
76  printf("Failed to get data from sensor\n");
77  return false;
78  }
79 
80  acc_integration_sleep_ms(1000 / DEFAULT_UPDATE_RATE_WAKEUP);
81  } while (!result.presence_detected);
82 
83  uint32_t detected_zone = (uint32_t)((float)(result.presence_distance - DEFAULT_START_M) / (float)DEFAULT_ZONE_LENGTH);
84  printf("Motion in zone: %u, distance: %d, score: %d\n", (unsigned int)detected_zone, (int)(result.presence_distance * 1000.0f),
85  (int)(result.presence_score * 1000.0f));
86 
87  acc_detector_presence_deactivate(handle);
88 
89  return true;
90 }

93 /**
94  * @brief Use the presence detector to track movement in front of the sensor
95  *
96  * @param[in] handle The presence detector handle
97  */
98 static bool execute_movement_tracking(acc_detector_presence_handle_t handle)
99 {
100  acc_detector_presence_result_t result;
101 
102  if (!acc_detector_presence_activate(handle))
103  {
104  printf("Failed to activate detector\n");
105  return false;
106  }
107 
108  do
109  {
110  if (!acc_detector_presence_get_next(handle, &result))
111  {
112  printf("Failed to get data from sensor\n");
113  return false;
114  }
115 
116  if (result.presence_detected)
117  {
118  uint32_t detected_zone = (uint32_t)((float)(result.presence_distance - DEFAULT_START_M) / (float)DEFAULT_ZONE_LENGTH);
119  printf("Motion in zone: %u, distance: %d, score: %d\n", (unsigned int)detected_zone,
120  (int)(result.presence_distance * 1000.0f),
121  (int)(result.presence_score * 1000.0f));
122  }
123 
124  acc_integration_sleep_ms(1000 / DEFAULT_UPDATE_RATE_TRACKING);
125  } while (result.presence_detected);
126 
127  printf("No motion, score: %d\n", (int)(result.presence_score * 1000.0f));
128 
129  acc_detector_presence_deactivate(handle);
130 
131  return true;
132 }

138 int acc_ref_app_smart_presence(int argc, char *argv[])
139 {
140  (void)argc;
141  (void)argv;
142  printf("Acconeer software version %s\n", acc_version_get());
143 
144  const acc_hal_t *hal = acc_hal_integration_get_implementation();
145 
146  if (!acc_rss_activate(hal))
147  {
148  printf("Failed to activate RSS\n");
149  return EXIT_FAILURE;
150  }
151 
152  acc_detector_presence_configuration_t presence_configuration = acc_detector_presence_configuration_create();
153  if (presence_configuration == NULL)
154  {
155  printf("Failed to create configuration\n");
156  acc_rss_deactivate();
157  return EXIT_FAILURE;
158  }
159 
160  set_default_configuration(presence_configuration);
161  acc_detector_presence_configuration_power_save_mode_set(presence_configuration, ACC_POWER_SAVE_MODE_OFF);
162 
163  acc_detector_presence_handle_t handle = acc_detector_presence_create(presence_configuration);
164  if (handle == NULL)
165  {
166  printf("Failed to create detector\n");
167  acc_detector_presence_configuration_destroy(&presence_configuration);
168  acc_detector_presence_destroy(&handle);
169  acc_rss_deactivate();
170  return EXIT_FAILURE;
171  }
172 
173  while (true)
174  {
175  if (!execute_wakeup(handle))
176  {
177  acc_detector_presence_configuration_destroy(&presence_configuration);
178  acc_detector_presence_destroy(&handle);
179  acc_rss_deactivate();
180  return EXIT_FAILURE;
181  }
182 
183  acc_detector_presence_configuration_update_rate_set(presence_configuration, DEFAULT_UPDATE_RATE_TRACKING);
184  acc_detector_presence_configuration_power_save_mode_set(presence_configuration, ACC_POWER_SAVE_MODE_SLEEP);
185 
186  if (!acc_detector_presence_reconfigure(&handle, presence_configuration))
187  {
188  printf("Failed to reconfigure detector\n");
189  acc_detector_presence_configuration_destroy(&presence_configuration);
190  acc_detector_presence_destroy(&handle);
191  acc_rss_deactivate();
192  return EXIT_FAILURE;
193  }
194 
195  if (!execute_movement_tracking(handle))
196  {
197  acc_detector_presence_configuration_destroy(&presence_configuration);
198  acc_detector_presence_destroy(&handle);
199  acc_rss_deactivate();
200  return EXIT_FAILURE;
201  }
202 
203  acc_detector_presence_configuration_update_rate_set(presence_configuration, DEFAULT_UPDATE_RATE_WAKEUP);
204  acc_detector_presence_configuration_power_save_mode_set(presence_configuration, ACC_POWER_SAVE_MODE_OFF);
205 
206  if (!acc_detector_presence_reconfigure(&handle, presence_configuration))
207  {
208  printf("Failed to reconfigure detector\n");
209  acc_detector_presence_configuration_destroy(&presence_configuration);
210  acc_detector_presence_destroy(&handle);
211  acc_rss_deactivate();
212  return EXIT_FAILURE;
213  }
214  }
215 
216  // We will never exit so no need to destroy the configuration or detector
217 
218  return EXIT_SUCCESS;
219 }

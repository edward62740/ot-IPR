18 /** \example example_detector_presence.c
19  * @brief This is an example on how the presence detector can be used
20  * @n
21  * The example executes as follows:
22  * - Activate Radar System Software (RSS)
23  * - Create a presence detector configuration
24  * - Create a presence detector using the previously created configuration
25  * - Destroy the presence detector configuration
26  * - Activate the presence detector
27  * - Get the result and print it 200 times
28  * - Deactivate and destroy the presence detector
29  * - Deactivate Radar System Software (RSS)
30  */

50 int acc_example_detector_presence(int argc, char *argv[])
51 {
52  (void)argc;
53  (void)argv;
54  printf("Acconeer software version %s\n", acc_version_get());
55 
56  const acc_hal_t *hal = acc_hal_integration_get_implementation();
57 
58  if (!acc_rss_activate(hal))
59  {
60  printf("Failed to activate RSS\n");
61  return EXIT_FAILURE;
62  }
63 
64  acc_detector_presence_configuration_t presence_configuration = acc_detector_presence_configuration_create();
65  if (presence_configuration == NULL)
66  {
67  printf("Failed to create configuration\n");
68  acc_rss_deactivate();
69  return EXIT_FAILURE;
70  }
71 
72  update_configuration(presence_configuration);
73 
74  acc_detector_presence_handle_t handle = acc_detector_presence_create(presence_configuration);
75  if (handle == NULL)
76  {
77  printf("Failed to create detector\n");
78  acc_detector_presence_configuration_destroy(&presence_configuration);
79  acc_rss_deactivate();
80  return EXIT_FAILURE;
81  }
82 
83  acc_detector_presence_configuration_destroy(&presence_configuration);
84 
85  if (!acc_detector_presence_activate(handle))
86  {
87  printf("Failed to activate detector\n");
88  acc_detector_presence_destroy(&handle);
89  acc_rss_deactivate();
90  return EXIT_FAILURE;
91  }
92 
93  bool success = true;
94  const int iterations = 200;
95  acc_detector_presence_result_t result;
96 
97  for (int i = 0; i < iterations; i++)
98  {
99  success = acc_detector_presence_get_next(handle, &result);
100  if (!success)
101  {
102  printf("acc_detector_presence_get_next() failed\n");
103  break;
104  }
105 
106  print_result(result);
107 
108  acc_integration_sleep_ms(1000 / DEFAULT_UPDATE_RATE);
109  }
110 
111  bool deactivated = acc_detector_presence_deactivate(handle);
112 
113  acc_detector_presence_destroy(&handle);
114 
115  acc_rss_deactivate();
116 
117  if (deactivated && success)
118  {
119  printf("Application finished OK\n");
120  return EXIT_SUCCESS;
121  }
122 
123  return EXIT_FAILURE;
124 }

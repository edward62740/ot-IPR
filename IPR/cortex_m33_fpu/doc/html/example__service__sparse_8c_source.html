21 /** \example example_service_sparse.c
22  * @brief This is an example on how the sparse service can be used
23  * @n
24  * The example executes as follows:
25  * - Activate Radar System Software (RSS)
26  * - Create a sparse service configuration
27  * - Create a sparse service using the previously created configuration
28  * - Destroy the sparse service configuration
29  * - Activate the sparse service
30  * - Get the result and print it 5 times
31  * - Deactivate and destroy the sparse service
32  * - Deactivate Radar System Software (RSS)
33  */

45 int acc_example_service_sparse(int argc, char *argv[])
46 {
47  (void)argc;
48  (void)argv;
49  printf("Acconeer software version %s\n", acc_version_get());
50 
51  const acc_hal_t *hal = acc_hal_integration_get_implementation();
52 
53  if (!acc_rss_activate(hal))
54  {
55  printf("acc_rss_activate() failed\n");
56  return EXIT_FAILURE;
57  }
58 
59  acc_service_configuration_t sparse_configuration = acc_service_sparse_configuration_create();
60 
61  if (sparse_configuration == NULL)
62  {
63  printf("acc_service_sparse_configuration_create() failed\n");
64  acc_rss_deactivate();
65  return EXIT_FAILURE;
66  }
67 
68  update_configuration(sparse_configuration);
69 
70  acc_service_handle_t handle = acc_service_create(sparse_configuration);
71 
72  if (handle == NULL)
73  {
74  printf("acc_service_create() failed\n");
75  acc_service_sparse_configuration_destroy(&sparse_configuration);
76  acc_rss_deactivate();
77  return EXIT_FAILURE;
78  }
79 
80  acc_service_sparse_metadata_t sparse_metadata = { 0 };
81  acc_service_sparse_get_metadata(handle, &sparse_metadata);
82 
83  uint16_t sweeps_per_frame = acc_service_sparse_configuration_sweeps_per_frame_get(sparse_configuration);
84 
85  printf("Start: %d mm\n", (int)(sparse_metadata.start_m * 1000.0f));
86  printf("Length: %u mm\n", (unsigned int)(sparse_metadata.length_m * 1000.0f));
87  printf("Data length: %u\n", (unsigned int)(sparse_metadata.data_length));
88  printf("Sweeps per frame: %u\n", (unsigned int)sweeps_per_frame);
89 
90  acc_service_sparse_configuration_destroy(&sparse_configuration);
91 
92  if (!acc_service_activate(handle))
93  {
94  printf("acc_service_activate() failed\n");
95  acc_service_destroy(&handle);
96  acc_rss_deactivate();
97  return EXIT_FAILURE;
98  }
99 
100  bool success = true;
101  const int iterations = 5;
102  uint16_t data[sparse_metadata.data_length];
103  acc_service_sparse_result_info_t result_info;
104 
105  for (int i = 0; i < iterations; i++)
106  {
107  success = acc_service_sparse_get_next(handle, data, sparse_metadata.data_length, &result_info);
108 
109  if (!success)
110  {
111  printf("acc_service_sparse_get_next() failed\n");
112  break;
113  }
114 
115  print_data(data, sparse_metadata.data_length, sweeps_per_frame);
116  }
117 
118  bool deactivated = acc_service_deactivate(handle);
119 
120  acc_service_destroy(&handle);
121 
122  acc_rss_deactivate();
123 
124  if (deactivated && success)
125  {
126  printf("Application finished OK\n");
127  return EXIT_SUCCESS;
128  }
129 
130  return EXIT_FAILURE;
131 }

147 void print_data(uint16_t *data, uint16_t data_length, uint16_t sweeps_per_frame)
148 {
149  uint16_t sweep_length = data_length / sweeps_per_frame;
150 
151  printf("Sparse data:\n");
152 
153  for (uint16_t sweep = 0; sweep < sweeps_per_frame; sweep++)
154  {
155  printf("Sweep %u:\t", (unsigned int)(sweep + 1));
156 
157  for (uint16_t sweep_idx = 0; sweep_idx < sweep_length; sweep_idx++)
158  {
159  printf("%6u", (unsigned int)data[(sweep * sweep_length) + sweep_idx]);
160  }
161 
162  printf("\n");
163  }

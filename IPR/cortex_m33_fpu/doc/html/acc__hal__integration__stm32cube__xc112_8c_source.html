70 static void acc_hal_integration_sensor_transfer(acc_sensor_id_t sensor_id, uint8_t *buffer, size_t buffer_size)
71 {
72  (void)sensor_id; // Ignore parameter sensor_id
73 
74  HAL_GPIO_WritePin(A111_SPI_SS_GPIO_Port, A111_SPI_SS_Pin, GPIO_PIN_RESET);
75 
76  HAL_SPI_TransmitReceive(&A111_SPI_HANDLE, buffer, buffer, buffer_size, 5000);

82 static void acc_hal_integration_sensor_power_on(acc_sensor_id_t sensor_id)
83 {
84  (void)sensor_id; // Ignore parameter sensor_id
85 
86  HAL_GPIO_WritePin(A111_SPI_SS_GPIO_Port, A111_SPI_SS_Pin, GPIO_PIN_SET);
87 
88  // Enable PMU and wait
89  HAL_GPIO_WritePin(XC112_PMU_ENABLE_GPIO_Port, XC112_PMU_ENABLE_Pin, GPIO_PIN_SET);
90  HAL_Delay(5);
91 
92  HAL_GPIO_WritePin(XC112_ENABLE_N_GPIO_Port, XC112_ENABLE_N_Pin, GPIO_PIN_RESET);
93  HAL_Delay(5);
94 
95  HAL_GPIO_WritePin(A111_ENABLE_GPIO_Port, A111_ENABLE_Pin, GPIO_PIN_SET);
96  HAL_Delay(5);
97 
98  // Select communication with only Sensor 1 and deselect all other radar slots
99  HAL_GPIO_WritePin(XC112_SPI_S1_ENABLE_N_GPIO_Port, XC112_SPI_S1_ENABLE_N_Pin, GPIO_PIN_RESET);
100  HAL_GPIO_WritePin(XC112_SPI_S2_ENABLE_N_GPIO_Port, XC112_SPI_S2_ENABLE_N_Pin, GPIO_PIN_SET);
101  HAL_GPIO_WritePin(XC112_SPI_S3_ENABLE_N_GPIO_Port, XC112_SPI_S3_ENABLE_N_Pin, GPIO_PIN_SET);
102  HAL_GPIO_WritePin(XC112_SPI_S4_ENABLE_N_GPIO_Port, XC112_SPI_S4_ENABLE_N_Pin, GPIO_PIN_SET);
103 }

117 static bool acc_hal_integration_wait_for_sensor_interrupt(acc_sensor_id_t sensor_id, uint32_t timeout_ms)
118 {
119  (void)sensor_id; // Ignore parameter sensor_id
120 
121  const uint32_t wait_begin_ms = HAL_GetTick();
122  while ((HAL_GPIO_ReadPin(A111_SENSOR_INTERRUPT_GPIO_Port, A111_SENSOR_INTERRUPT_Pin) != GPIO_PIN_SET) &&
123  (HAL_GetTick() - wait_begin_ms < timeout_ms))
124  {
125  // Wait for the GPIO interrupt
126  disable_interrupts();
127  // Check again so that IRQ did not occur
128  if (HAL_GPIO_ReadPin(A111_SENSOR_INTERRUPT_GPIO_Port, A111_SENSOR_INTERRUPT_Pin) != GPIO_PIN_SET)
129  {
130  __WFI();
131  }
132 
133  // Enable interrupts again to allow pending interrupt to be handled
134  enable_interrupts();
135  }
136 
137  return HAL_GPIO_ReadPin(A111_SENSOR_INTERRUPT_GPIO_Port, A111_SENSOR_INTERRUPT_Pin) == GPIO_PIN_SET;
138 }

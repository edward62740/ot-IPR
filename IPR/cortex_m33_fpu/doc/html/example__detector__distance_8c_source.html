43

45 
46  if (!acc_rss_activate(hal))

48  printf("acc_rss_activate() failed\n");
49  return EXIT_FAILURE;

54  if (distance_configuration == NULL)

56  printf("acc_detector_distance_configuration_create() failed\n");
57  acc_rss_deactivate();
58  return EXIT_FAILURE;
59  }
60 
61  acc_detector_distance_handle_t distance_handle = acc_detector_distance_create(distance_configuration);
62 
63  if (distance_handle == NULL)
64  {
65  printf("acc_detector_distance_create() failed\n");
66  acc_detector_distance_configuration_destroy(&distance_configuration);
67  acc_rss_deactivate();
68  return EXIT_FAILURE;
69  }
70 
71  acc_detector_distance_configuration_destroy(&distance_configuration);
72 
73  if (!acc_detector_distance_activate(distance_handle))
74  {
75  printf("acc_detector_distance_activate() failed\n");
76  acc_detector_distance_destroy(&distance_handle);
77  acc_rss_deactivate();
78  return EXIT_FAILURE;
79  }
80 
81  bool success = true;
82  const int iterations = 5;
83  uint16_t number_of_peaks = 5;
84  acc_detector_distance_result_t result[number_of_peaks];
85  acc_detector_distance_result_info_t result_info;
86 
87  for (int i = 0; i < iterations; i++)
88  {
89  success = acc_detector_distance_get_next(distance_handle, result, number_of_peaks, &result_info);
90 
91  if (!success)

93  printf("acc_detector_distance_get_next() failed\n");
94  break;

96 
97  print_distances(result, result_info.number_of_peaks);
98  }
99 
100  bool deactivated = acc_detector_distance_deactivate(distance_handle);
101 
102  acc_detector_distance_destroy(&distance_handle);
103 
104  acc_rss_deactivate();
105 
106  if (deactivated && success)
107  {
108  printf("Application finished OK\n");
109  return EXIT_SUCCESS;
110  }
111 
112  return EXIT_FAILURE;
113 }

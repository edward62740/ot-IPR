106 /**
107  * @brief Definition of a sensor power function
108  *
109  * These functions control the power of the sensor. It typically control PS_ENABLE
110  * and PMU_ENABLE. The hibernate functions should also toggle the CTRL pin.
111  *
112  * In the case of the power_on function:
113  * Any pending sensor interrupts should be cleared before returning from function.
114  */
115 typedef void (*acc_hal_sensor_power_function_t)(acc_sensor_id_t sensor_id);

127 /**
128  * @brief Definition of a wait for sensor interrupt function
129  *
130  * This function shall wait at most timeout_ms for the interrupt to become active and
131  * then return true. It may return true immediately if an interrupt have
132  * occurred since last call to this function.
133  *
134  * If waited more than timeout_ms for the interrupt to become active it shall
135  * return false.
136  *
137  * Note that this function can be called with a timeout_ms = 0.
138  *
139  */
140 typedef bool (*acc_hal_sensor_wait_for_interrupt_function_t)(acc_sensor_id_t sensor_id, uint32_t timeout_ms);

143 /**
144  * @brief Definition of a sensor transfer function
145  *
146  * This function shall transfer data to and from the sensor over spi. It's beneficial from a performance perspective
147  * to use dma if available.
148  * The buffer is naturally aligned to a maximum of 4 bytes.
149  *
150  */
151 typedef void (*acc_hal_sensor_transfer_function_t)(acc_sensor_id_t sensor_id, uint8_t *buffer, size_t buffer_size);

10 /**
11  * @brief Power save mode
12  *
13  * Each power save mode corresponds to how much of the sensor hardware is shutdown
14  * between data frame aquisition. Mode 'OFF' means that the whole sensor is shutdown.
15  * Mode 'ACTIVE' means that the sensor is in its active state all the time.
16  * Mode 'HIBERNATE' means that the sensor is still powered but the internal oscillator is
17  * turned off and the application needs to clock the sensor by toggling a GPIO a pre-defined
18  * number of times to enter and exit this mode. Mode 'HIBERNATE' is currently only supported
19  * by the Sparse service
20  *
21  * For each power save mode there will be a limit in the achievable update rate. Mode 'OFF'
22  * can achieve the lowest power consumption but also has the lowest update rate limit
23  */
24 typedef enum
25 {
26  ACC_POWER_SAVE_MODE_OFF,
27  ACC_POWER_SAVE_MODE_SLEEP,
28  ACC_POWER_SAVE_MODE_READY,
29  ACC_POWER_SAVE_MODE_ACTIVE,
30  ACC_POWER_SAVE_MODE_HIBERNATE,
31 } acc_power_save_mode_enum_t;

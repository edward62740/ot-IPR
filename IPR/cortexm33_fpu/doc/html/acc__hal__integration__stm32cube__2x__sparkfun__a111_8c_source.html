63 static void acc_hal_integration_sensor_transfer(acc_sensor_id_t sensor_id, uint8_t *buffer, size_t buffer_size)
64 {
65  uint32_t sensor_cs_pin;
66  GPIO_TypeDef *sensor_cs_port;
67 
68  if ((sensor_id == 0) || (sensor_id > SENSOR_COUNT))
69  {
70  return;
71  }
72 
73  switch (sensor_id)
74  {
75  case 1:
76  sensor_cs_pin = A111_ID1_SPI_SS_N_Pin;
77  sensor_cs_port = A111_ID1_SPI_SS_N_GPIO_Port;
78  break;
79  case 2:
80  sensor_cs_pin = A111_ID2_SPI_SS_N_Pin;
81  sensor_cs_port = A111_ID2_SPI_SS_N_GPIO_Port;
82  break;
83  default:
84  return;
85  }
86 
87  const uint32_t SPI_TRANSMIT_RECEIVE_TIMEOUT = 5000;
88 
89  HAL_GPIO_WritePin(sensor_cs_port, sensor_cs_pin, GPIO_PIN_RESET);
90  HAL_SPI_TransmitReceive(&A111_SPI_HANDLE, buffer, buffer, buffer_size, SPI_TRANSMIT_RECEIVE_TIMEOUT);
91  HAL_GPIO_WritePin(sensor_cs_port, sensor_cs_pin, GPIO_PIN_SET);
92 }

95 static void acc_hal_integration_sensor_power_on(acc_sensor_id_t sensor_id)
96 {
97  if ((sensor_id == 0) || (sensor_id > SENSOR_COUNT))
98  {
99  return;
100  }
101 
102  switch (sensor_id)
103  {
104  case 1:
105  HAL_GPIO_WritePin(A111_ID1_SPI_SS_N_GPIO_Port, A111_ID1_SPI_SS_N_Pin, GPIO_PIN_SET);
106  HAL_GPIO_WritePin(A111_ID1_ENABLE_GPIO_Port, A111_ID1_ENABLE_Pin, GPIO_PIN_SET);
107  break;
108  case 2:
109  HAL_GPIO_WritePin(A111_ID2_SPI_SS_N_GPIO_Port, A111_ID2_SPI_SS_N_Pin, GPIO_PIN_SET);
110  HAL_GPIO_WritePin(A111_ID2_ENABLE_GPIO_Port, A111_ID2_ENABLE_Pin, GPIO_PIN_SET);
111  break;
112  default:
113  break;
114  }
115 
116  // Wait 2 ms to make sure that the sensor crystal have time to stabilize
117  HAL_Delay(2);
118 }

121 static void acc_hal_integration_sensor_power_off(acc_sensor_id_t sensor_id)
122 {
123  if ((sensor_id == 0) || (sensor_id > SENSOR_COUNT))
124  {
125  return;
126  }
127 
128  switch (sensor_id)
129  {
130  case 1:
131  HAL_GPIO_WritePin(A111_ID1_ENABLE_GPIO_Port, A111_ID1_ENABLE_Pin, GPIO_PIN_RESET);
132  break;
133  case 2:
134  HAL_GPIO_WritePin(A111_ID2_ENABLE_GPIO_Port, A111_ID2_ENABLE_Pin, GPIO_PIN_RESET);
135  break;
136  default:
137  break;
138  }
139 
140  // Wait after power off to leave the sensor in a known state
141  // in case the application intends to enable the sensor directly
142  HAL_Delay(2);
143 }

146 static bool acc_hal_integration_wait_for_sensor_interrupt(acc_sensor_id_t sensor_id, uint32_t timeout_ms)
147 {
148  uint32_t sensor_interrupt_pin;
149  GPIO_TypeDef *sensor_interrupt_port;
150 
151  if ((sensor_id == 0) || (sensor_id > SENSOR_COUNT))
152  {
153  return false;
154  }
155 
156  switch (sensor_id)
157  {
158  case 1:
159  sensor_interrupt_pin = A111_ID1_SENSOR_INTERRUPT_Pin;
160  sensor_interrupt_port = A111_ID1_SENSOR_INTERRUPT_GPIO_Port;
161  break;
162  case 2:
163  sensor_interrupt_pin = A111_ID2_SENSOR_INTERRUPT_Pin;
164  sensor_interrupt_port = A111_ID2_SENSOR_INTERRUPT_GPIO_Port;
165  break;
166  default:
167  return false;
168  }
169 
170  const uint32_t wait_begin_ms = HAL_GetTick();
171  while ((HAL_GPIO_ReadPin(sensor_interrupt_port, sensor_interrupt_pin) != GPIO_PIN_SET) &&
172  (HAL_GetTick() - wait_begin_ms < timeout_ms))
173  {
174  // Wait for the GPIO interrupt
175  disable_interrupts();
176  // Check again so that IRQ did not occur
177  if (HAL_GPIO_ReadPin(sensor_interrupt_port, sensor_interrupt_pin) != GPIO_PIN_SET)
178  {
179  __WFI();
180  }
181 
182  // Enable interrupts again to allow pending interrupt to be handled
183  enable_interrupts();
184  }
185 
186  return HAL_GPIO_ReadPin(sensor_interrupt_port, sensor_interrupt_pin) == GPIO_PIN_SET;
187 }

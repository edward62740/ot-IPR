19 /** \example example_multiple_service_usage.c
20  * @brief This is an example on how the RSS configuration 'override sensor id check at creation' can be used for using multiple services
21  *
22  * It is mainly a tool to easier and faster switch between services and detectors
23  * on the same sensor. This can also be used to switch between configurations for the
24  * same service or detector.
25  *
26  * @n
27  * The example executes as follows:
28  * - Activate Radar System Software (RSS)
29  * - Set 'override sensor id check at creation' to true
30  * - Create an envelope service configuration
31  * - Create a sparse service configuration
32  * - Create an envelope service using the envelope service configuration
33  * - Create a sparse service using the sparse service configuration
34  * - Destroy the configurations
35  * - In a loop:
36  * - Activate the envelope service
37  * - Get the result and print it
38  * - Deactivate the envelope service
39  * - Activate the sparse service
40  * - Get the result and print it
41  * - Deactivate the sparse service
42  * - End of loop
43  * - Destroy the services
44  * - Deactivate Radar System Software (RSS)
45  */

57 int acc_example_multiple_service_usage(int argc, char *argv[])
58 {
59  (void)argc;
60  (void)argv;
61  printf("Acconeer software version %s\n", acc_version_get());
62 
63  const acc_hal_t *hal = acc_hal_integration_get_implementation();
64 
65  // Activate RSS and override sensor ID check
66  if (!acc_rss_activate(hal))
67  {
68  printf("acc_rss_activate() failed\n");
69  return EXIT_FAILURE;
70  }
71 
72  acc_rss_override_sensor_id_check_at_creation(true);
73 
74  // Create service configurations
75  acc_service_configuration_t envelope_configuration = acc_service_envelope_configuration_create();
76 
77  if (envelope_configuration == NULL)
78  {
79  printf("acc_service_envelope_configuration_create() failed\n");
80  acc_rss_deactivate();
81  return EXIT_FAILURE;
82  }
83 
84  acc_service_configuration_t sparse_configuration = acc_service_sparse_configuration_create();
85 
86  if (sparse_configuration == NULL)
87  {
88  printf("acc_service_sparse_configuration_create() failed\n");
89  acc_rss_deactivate();
90  return EXIT_FAILURE;
91  }
92 
93  // Create envelope service and extract metadata
94  acc_service_handle_t envelope_handle = acc_service_create(envelope_configuration);
95 
96  if (envelope_handle == NULL)
97  {
98  printf("acc_service_create() for envelope failed\n");
99  acc_service_envelope_configuration_destroy(&envelope_configuration);
100  acc_service_sparse_configuration_destroy(&sparse_configuration);
101  acc_rss_deactivate();
102  return EXIT_FAILURE;
103  }
104 
105  acc_service_envelope_metadata_t envelope_metadata = { 0 };
106  acc_service_envelope_get_metadata(envelope_handle, &envelope_metadata);
107 
108  acc_service_envelope_configuration_destroy(&envelope_configuration);
109 
110  // Create sparse service and extract metadata
111  acc_service_handle_t sparse_handle = acc_service_create(sparse_configuration);
112 
113  if (sparse_handle == NULL)
114  {
115  printf("acc_service_create() for sparse failed\n");
116  acc_service_destroy(&envelope_handle);
117  acc_service_sparse_configuration_destroy(&sparse_configuration);
118  acc_rss_deactivate();
119  return EXIT_FAILURE;
120  }
121 
122  acc_service_sparse_metadata_t sparse_metadata = { 0 };
123  acc_service_sparse_get_metadata(sparse_handle, &sparse_metadata);
124 
125  uint16_t sweeps_per_frame = acc_service_sparse_configuration_sweeps_per_frame_get(sparse_configuration);
126 
127  acc_service_sparse_configuration_destroy(&sparse_configuration);
128 
129  // Activate, retrieve result and deactivate the services
130  bool success = true;
131  const int iterations = 2;
132 
133  uint16_t *envelope_data;
134  acc_service_envelope_result_info_t envelope_result_info;
135 
136  uint16_t *sparse_data;
137  acc_service_sparse_result_info_t sparse_result_info;
138 
139  for (int i = 0; i < iterations; i++)
140  {
141  // Envelope service
142  if (!acc_service_activate(envelope_handle))
143  {
144  success = false;
145  printf("acc_service_activate() for envelope failed\n");
146  break;
147  }
148 
149  if (!acc_service_envelope_get_next_by_reference(envelope_handle, &envelope_data, &envelope_result_info))
150  {
151  acc_service_deactivate(envelope_handle);
152  success = false;
153  printf("acc_service_envelope_get_next_by_reference() failed\n");
154  break;
155  }
156 
157  print_envelope_data(envelope_data, envelope_metadata.data_length);
158 
159  if (!acc_service_deactivate(envelope_handle))
160  {
161  success = false;
162  printf("acc_service_deactivate() for envelope failed\n");
163  break;
164  }
165 
166  // Sparse service
167  if (!acc_service_activate(sparse_handle))
168  {
169  success = false;
170  printf("acc_service_activate() for sparse failed\n");
171  break;
172  }
173 
174  if (!acc_service_sparse_get_next_by_reference(sparse_handle, &sparse_data, &sparse_result_info))
175  {
176  acc_service_deactivate(sparse_handle);
177  success = false;
178  printf("acc_service_sparse_get_next_by_reference() failed\n");
179  break;
180  }
181 
182  print_sparse_data(sparse_data, sparse_metadata.data_length, sweeps_per_frame);
183 
184  if (!acc_service_deactivate(sparse_handle))
185  {
186  success = false;
187  printf("acc_service_deactivate() for sparse failed\n");
188  break;
189  }
190  }
191 
192  acc_service_destroy(&envelope_handle);
193  acc_service_destroy(&sparse_handle);
194 
195  if (success)
196  {
197  printf("Application finished OK\n");
198  return EXIT_SUCCESS;
199  }
200 
201  return EXIT_FAILURE;
202 }

205 void print_envelope_data(uint16_t *data, uint16_t data_length)
206 {
207  printf("Envelope data:\n");
208  for (uint16_t i = 0; i < data_length; i++)
209  {
210  if ((i > 0) && ((i % 8) == 0))
211  {
212  printf("\n");
213  }
214 
215  printf("%6u", (unsigned int)(data[i]));
216  }
217 
218  printf("\n\n");
219 }

222 void print_sparse_data(uint16_t *data, uint16_t data_length, uint16_t sweeps_per_frame)
223 {
224  uint16_t sweep_length = data_length / sweeps_per_frame;
225 
226  printf("Sparse data:\n");
227  for (uint16_t sweep = 0; sweep < sweeps_per_frame; sweep++)
228  {
229  printf("Sweep %u:\t", (unsigned int)(sweep + 1));
230 
231  for (uint16_t sweep_idx = 0; sweep_idx < sweep_length; sweep_idx++)
232  {
233  printf("%6u", (unsigned int)data[(sweep * sweep_length) + sweep_idx]);
234  }
235 
236  printf("\n");
237  }
238 
239  printf("\n");
240 }

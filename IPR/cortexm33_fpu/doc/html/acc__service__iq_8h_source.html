96 /**
97  * @brief Get the distance domain lowpass filter cutoff frequency ratio override parameters
98  *
99  * The cutoff for the distance domain lowpass filter is specified as the ratio between the spatial
100  * frequency cutoff and the sample frequency. A ratio close to zero damps everything except the lowest
101  * frequencies. Increasing ratios output a wider band of spatial frequencies, and a ratio of 0.5 means
102  * that the filter is deactivated.
103  *
104  * If unset, i.e., if override is false, the ratio will be chosen automatically. This ratio is
105  * returned in the metadata upon creation of the service.
106  *
107  * @param[in] service_configuration The configuration to get the parameters from
108  * @param[out] override If true, the specified cutoff ratio is used
109  * @param[out] cutoff_ratio The cutoff ratio to use if override is true
110  */
111 void acc_service_iq_depth_lowpass_cutoff_ratio_get(acc_service_configuration_t service_configuration, bool *override, float *cutoff_ratio);
112 
113 
114 /**
115  * @brief Set the distance domain lowpass filter cutoff frequency ratio override parameters
116  *
117  * The cutoff for the distance domain lowpass filter is specified as the ratio between the spatial
118  * frequency cutoff and the sample frequency. An input value of zero for the cutoff ratio will
119  * configure the smoothest allowed filter. A cutoff ratio of 0.5 turns the filter off.
120  *
121  * The set of available cutoff frequencies is limited due to internal properties of the filter
122  * implementation.
123  *
124  * If unset, i.e., if override is false, the ratio will be chosen automatically. This ratio is
125  * returned in the metadata upon creation of the service.
126  *
127  * @param[in] service_configuration The configuration to set the parameters for
128  * @param[in] override If true, the specified cutoff ratio is used
129  * @param[in] cutoff_ratio The cutoff ratio to use if override is true
130  */
131 void acc_service_iq_depth_lowpass_cutoff_ratio_set(acc_service_configuration_t service_configuration,
132  bool override,
133  float cutoff_ratio);

136 /**
137  * @brief Get the sensor downsampling factor
138  *
139  * Gets the downsampling factor - the number of steps taken between each data point. A downsampling factor of 1 samples
140  * every possible point in the range. A downsampling factor of 2 samples every other point, and so on.
141  *
142  * In the IQ service, the base step length is ~0.5mm. Thus, for example setting downsampling factor to 4 makes
143  * the distance between two points in the measured range ~2mm.
144  *
145  * @param[in] service_configuration The configuration to get downsampling factor from
146  * @return downsampling_factor The downsampling factor
147  */
148 uint16_t acc_service_iq_downsampling_factor_get(acc_service_configuration_t service_configuration);

155  * every possible point in the range. A downsampling factor of 2 samples every other point, and so on.
156  *
157  * In the IQ service, the base step length is ~0.5mm. Thus, for example setting downsampling factor to 4 makes

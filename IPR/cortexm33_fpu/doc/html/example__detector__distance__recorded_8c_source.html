17 /** \example example_detector_distance_recorded.c
18  *
19  * @brief This is an example on how the distance detector can be used when using recorded threshold type.
20  *
21  * @n
22  * It shows both how to record a background as well as setting a previously recorded background.
23  *
24  * @n
25  * The example executes as follows:
26  * - Activate Radar System Software (RSS)
27  * - Create a distance detector configuration
28  * - Create a distance detector using the previously created configuration
29  * - Record a background
30  * - Deactivate and destroy the distance detector
31  * - Create a distance detector using the previously created configuration
32  * - Set the previously recorded background
33  * - Activate the distance detector
34  * - Get the result and print it 5 times
35  * - Deactivate and destroy the distance detector
36  * - Destroy the distance detector configuration
37  * - Deactivate Radar System Software (RSS)
38  */

59 int acc_example_detector_distance_recorded(int argc, char *argv[])
60 {
61  (void)argc;
62  (void)argv;
63  printf("Acconeer software version %s\n", acc_version_get());
64 
65  const acc_hal_t *hal = acc_hal_integration_get_implementation();
66 
67  if (!acc_rss_activate(hal))
68  {
69  printf("acc_rss_activate() failed\n");
70  return EXIT_FAILURE;
71  }
72 
73  acc_detector_distance_configuration_t distance_configuration = acc_detector_distance_configuration_create();
74 
75  if (distance_configuration == NULL)
76  {
77  printf("acc_detector_distance_configuration_create() failed\n");
78  acc_rss_deactivate();
79  return EXIT_FAILURE;
80  }
81 
82  acc_detector_distance_configuration_threshold_type_set(distance_configuration, ACC_DETECTOR_DISTANCE_THRESHOLD_TYPE_RECORDED);
83  acc_detector_distance_configuration_record_background_sweeps_set(distance_configuration, 10);
84  acc_detector_distance_configuration_requested_start_set(distance_configuration, 0.2f);
85  acc_detector_distance_configuration_requested_length_set(distance_configuration, 0.5f);
86 
87  if (!record_background(distance_configuration))
88  {
89  acc_detector_distance_configuration_destroy(&distance_configuration);
90  acc_rss_deactivate();
91  return EXIT_FAILURE;
92  }
93 
94  if (!measure_with_recorded_background(distance_configuration))
95  {
96  acc_detector_distance_configuration_destroy(&distance_configuration);
97  acc_rss_deactivate();
98  return EXIT_FAILURE;
99  }
100 
101  acc_detector_distance_configuration_destroy(&distance_configuration);
102 
103  acc_rss_deactivate();
104 
105  printf("Application finished OK\n");
106  return EXIT_SUCCESS;
107 }

110 bool record_background(acc_detector_distance_configuration_t distance_configuration)
111 {
112  acc_detector_distance_handle_t distance_handle = acc_detector_distance_create(distance_configuration);
113 
114  if (distance_handle == NULL)
115  {
116  printf("acc_detector_distance_create() failed\n");
117  return false;
118  }
119 
120  acc_detector_distance_metadata_t metadata;
121  acc_detector_distance_metadata_get(distance_handle, &metadata);
122 
123  background_length = metadata.background_length;
124 
125  if (background_length > MAX_BACKGROUND_LENGTH)
126  {
127  printf("background_length larger than MAX_BACKGROUND_LENGTH\n");
128  acc_detector_distance_destroy(&distance_handle);
129  return false;
130  }
131 
132  if (!acc_detector_distance_record_background(distance_handle, background, background_length, NULL))
133  {
134  printf("acc_detector_distance_record_background() failed\n");
135  acc_detector_distance_destroy(&distance_handle);
136  return false;
137  }
138 
139  acc_detector_distance_destroy(&distance_handle);
140 
141  return true;
142 }

145 bool measure_with_recorded_background(acc_detector_distance_configuration_t distance_configuration)
146 {
147  acc_detector_distance_handle_t distance_handle = acc_detector_distance_create(distance_configuration);
148 
149  if (distance_handle == NULL)
150  {
151  printf("acc_detector_distance_create() failed\n");
152  return false;
153  }
154 
155  if (!acc_detector_distance_set_background(distance_handle, background, background_length))
156  {
157  printf("acc_detector_distance_set_background() failed\n");
158  acc_detector_distance_destroy(&distance_handle);
159  return false;
160  }
161 
162  if (!acc_detector_distance_activate(distance_handle))
163  {
164  printf("acc_detector_distance_activate() failed\n");
165  acc_detector_distance_destroy(&distance_handle);
166  return false;
167  }
168 
169  bool success = true;
170  const int iterations = 5;
171  uint16_t number_of_peaks = 5;
172  acc_detector_distance_result_t result[number_of_peaks];
173  acc_detector_distance_result_info_t result_info;
174 
175  for (int i = 0; i < iterations; i++)
176  {
177  success = acc_detector_distance_get_next(distance_handle, result, number_of_peaks, &result_info);
178 
179  if (!success)
180  {
181  printf("acc_detector_distance_get_next() failed\n");
182  break;
183  }
184 
185  print_distances(result, result_info.number_of_peaks);
186  }
187 
188  bool deactivated = acc_detector_distance_deactivate(distance_handle);
189 
190  acc_detector_distance_destroy(&distance_handle);
191 
192  return deactivated && success;
193 }
